// ===== 坦克大战：核心逻辑 ===== function initTankGame(playerName) { if (!gameCanvas) return; ctx = gameCanvas.getContext("2d"); const cw = gameCanvas.width; const ch = gameCanvas.height; tankPlayer = { x: cw / 2, y: ch - 80, size: 32, dir: "up", speed: 3, fireCooldown: 0 }; tankEnemies = []; tankBullets = []; tankEnemyBullets = []; tankSpawnTimer = 0; tankLives = 3; tankScore = 0; tankGameOver = false; hudPlayer.textContent = "玩家：" + (playerName || "游客"); hudLap.textContent = `得分：${tankScore}`; hudTime.textContent = `生命：${tankLives}`; hudRank.textContent = `波次：1`; raceBanner.style.display = "none"; tankRunning = true; tankLastTime = 0; requestAnimationFrame(tankGameLoop); } function spawnEnemyTank() { const cw = gameCanvas.width; const ch = gameCanvas.height; const size = 32; const spawnXs = [cw * 0.2, cw * 0.5, cw * 0.8]; const x = spawnXs[Math.floor(Math.random() * spawnXs.length)]; tankEnemies.push({ x, y: 80, size, dir: "down", speed: 1.6 + Math.random() * 0.6, fireCooldown: 1 + Math.random() * 1.5 }); } function fireBullet(owner, isEnemy) { const speed = isEnemy ? 4 : 6; let dx = 0; let dy = 0; if (owner.dir === "up") dy = -1; else if (owner.dir === "down") dy = 1; else if (owner.dir === "left") dx = -1; else if (owner.dir === "right") dx = 1; const arr = isEnemy ? tankEnemyBullets : tankBullets; arr.push({ x: owner.x, y: owner.y, dx: dx * speed, dy: dy * speed, size: 6 }); } function updateTankWorld(dt, dtSeconds) { if (!tankPlayer) return; const cw = gameCanvas.width; const ch = gameCanvas.height; const up = keys["ArrowUp"] || keys["w"]; const down = keys["ArrowDown"] || keys["s"]; const left = keys["ArrowLeft"] || keys["a"]; const right = keys["ArrowRight"] || keys["d"]; const fire = keys[" "]; // 玩家移动与朝向 if (up) { tankPlayer.y -= tankPlayer.speed * dt; tankPlayer.dir = "up"; } else if (down) { tankPlayer.y += tankPlayer.speed * dt; tankPlayer.dir = "down"; } else if (left) { tankPlayer.x -= tankPlayer.speed * dt; tankPlayer.dir = "left"; } else if (right) { tankPlayer.x += tankPlayer.speed * dt; tankPlayer.dir = "right"; } // 边界限制 const margin = 24; if (tankPlayer.x < margin) tankPlayer.x = margin; if (tankPlayer.x > cw - margin) tankPlayer.x = cw - margin; if (tankPlayer.y < margin + 40) tankPlayer.y = margin + 40; if (tankPlayer.y > ch - margin) tankPlayer.y = ch - margin; // 玩家开火冷却 if (tankPlayer.fireCooldown > 0) { tankPlayer.fireCooldown -= dtSeconds; } if (fire && tankPlayer.fireCooldown <= 0 && !tankGameOver) { fireBullet(tankPlayer, false); tankPlayer.fireCooldown = 0.35; } // 敌人生成 tankSpawnTimer += dtSeconds; if (tankSpawnTimer > 2 && tankEnemies.length < 5 && !tankGameOver) { spawnEnemyTank(); tankSpawnTimer = 0; } // 敌人移动与开火 for (const e of tankEnemies) { // 简单 AI：向玩家靠近 if (Math.abs(e.x - tankPlayer.x) > 12) { if (tankPlayer.x < e.x) { e.x -= e.speed * dt; e.dir = "left"; } else { e.x += e.speed * dt; e.dir = "right"; } } else { if (tankPlayer.y < e.y) { e.y -= e.speed * dt; e.dir = "up"; } else { e.y += e.speed * dt; e.dir = "down"; } } // 敌人开火 if (e.fireCooldown > 0) { e.fireCooldown -= dtSeconds; } else if (!tankGameOver) { fireBullet(e, true); e.fireCooldown = 1 + Math.random() * 1.5; } } // 子弹更新 const inBounds = (b) => b.x > -20 && b.x < cw + 20 && b.y > -20 && b.y < ch + 20; tankBullets.forEach((b) => { b.x += b.dx * dt; b.y += b.dy * dt; }); tankEnemyBullets.forEach((b) => { b.x += b.dx * dt; b.y += b.dy * dt; }); tankBullets = tankBullets.filter(inBounds); tankEnemyBullets = tankEnemyBullets.filter(inBounds); // 玩家子弹击中敌人 for (let i = tankBullets.length - 1; i >= 0; i--) { const b = tankBullets[i]; let hitIndex = -1; for (let j = 0; j < tankEnemies.length; j++) { const e = tankEnemies[j]; const half = e.size / 2; if ( b.x > e.x - half && b.x < e.x + half && b.y > e.y - half && b.y < e.y + half ) { hitIndex = j; break; } } if (hitIndex !== -1) { tankEnemies.splice(hitIndex, 1); tankBullets.splice(i, 1); tankScore += 100; hudLap.textContent = `得分：${tankScore}`; } } // 敌人子弹击中玩家 if (!tankGameOver) { for (let i = tankEnemyBullets.length - 1; i >= 0; i--) { const b = tankEnemyBullets[i]; const half = tankPlayer.size / 2; if ( b.x > tankPlayer.x - half && b.x < tankPlayer.x + half && b.y > tankPlayer.y - half && b.y < tankPlayer.y + half ) { tankEnemyBullets.splice(i, 1); tankLives -= 1; hudTime.textContent = `生命：${tankLives}`; if (tankLives <= 0) { tankGameOver = true; raceBanner.style.display = "block"; raceBanner.textContent = `基地被摧毁 · 得分 ${tankScore}`; } } } } } function drawTankGame() { const cw = gameCanvas.width; const ch = gameCanvas.height; ctx.clearRect(0, 0, cw, ch); ctx.fillStyle = "#020617"; ctx.fillRect(0, 0, cw, ch); // 背景网格 ctx.strokeStyle = "rgba(30, 64, 175, 0.25)"; ctx.lineWidth = 1; const gridSize = 24; ctx.stroke(); } // 玩家坦克 if (tankPlayer) { drawTankSprite(tankPlayer, true); } // 敌人坦克 for (const e of tankEnemies) { drawTankSprite(e, false); } // 子弹 ctx.fillStyle = "#e5e7eb"; for (const b of tankBullets) { ctx.beginPath(); ctx.arc(b.x, b.y, b.size / 2, 0, Math.PI * 2); ctx.fill(); } ctx.fillStyle = "#f97373"; for (const b of tankEnemyBullets) { ctx.beginPath(); ctx.arc(b.x, b.y, b.size / 2, 0, Math.PI * 2); ctx.fill(); } } function drawTankSprite(t, isPlayer) { const size = t.size; const half = size / 2; ctx.save(); ctx.translate(t.x, t.y); let angle = 0; if (t.dir === "up") angle = 0; else if (t.dir === "right") angle = Math.PI / 2; else if (t.dir === "down") angle = Math.PI; else if (t.dir === "left") angle = -Math.PI / 2; ctx.rotate(angle); // 底座 ctx.fillStyle = isPlayer ? "#22d3ee" : "#f97316"; ctx.fillRect(-half, -half, size, size); // 履带 ctx.fillStyle = "#020617"; ctx.fillRect(-half - 4, -half, 6, size); ctx.fillRect(half - 2, -half, 6, size); // 车体 ctx.fillStyle = isPlayer ? "#0f172a" : "#111827"; ctx.fillRect(-half * 0.6, -half * 0.6, size * 0.6, size * 0.6); // 炮塔 ctx.fillStyle = "#e5e7eb"; ctx.beginPath(); ctx.arc(0, 0, size * 0.18, 0, Math.PI * 2); ctx.fill(); // 炮管 ctx.strokeStyle = "#e5e7eb"; ctx.lineWidth = 4; ctx.beginPath(); ctx.moveTo(0, 0); ctx.lineTo(0, -half - 6); ctx.stroke(); ctx.restore(); } function tankGameLoop(timestamp) { if (!tankRunning) return; if (!tankLastTime) tankLastTime = timestamp; const deltaMs = timestamp - tankLastTime; tankLastTime = timestamp; const dt = Math.min(deltaMs / 16.67, 2); const dtSeconds = deltaMs / 1000; updateTankWorld(dt, dtSeconds); drawTankGame(); requestAnimationFrame(tankGameLoop); }
